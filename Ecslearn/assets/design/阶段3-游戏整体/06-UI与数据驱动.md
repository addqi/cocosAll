## 3. UI 管理器

> 参考：幽灵射手的 `UIManager` 按路径懒加载预制体，我们简化它。

```typescript
class UIManager {
    private _cache: Map<string, Node> = new Map();
    private _popupRoot: Node;

    async show<T extends UIPanel>(
        panelClass: Constructor<T>,
        data?: any
    ): Promise<T> {
        const name = panelClass.name;

        let node = this._cache.get(name);
        if (!node) {
            const prefab = await loadPrefab(`ui/${name}`);
            node = instantiate(prefab);
            this._cache.set(name, node);
        }

        this._popupRoot.addChild(node);
        node.active = true;

        const panel = node.getComponent(panelClass)!;
        panel.show(data);
        return panel;
    }

    hide(panelClass: Constructor<UIPanel>): void {
        const name = panelClass.name;
        const node = this._cache.get(name);
        if (node) {
            node.active = false;
            node.removeFromParent();
        }
    }
}

abstract class UIPanel extends Component {
    abstract show(data?: any): void;
    abstract hide(): void;
}
```

## 4. 虚拟摇杆

```typescript
class Joystick extends Component {
    private _center: Vec2 = Vec2.ZERO;
    private _maxRadius: number = 60;
    private _moveDir: Vec2 = new Vec2();
    private _isActive: boolean = false;

    onEnable(): void {
        this.node.on(Node.EventType.TOUCH_START, this._onTouchStart, this);
        this.node.on(Node.EventType.TOUCH_MOVE, this._onTouchMove, this);
        this.node.on(Node.EventType.TOUCH_END, this._onTouchEnd, this);
        this.node.on(Node.EventType.TOUCH_CANCEL, this._onTouchEnd, this);
    }

    private _onTouchStart(event: EventTouch): void {
        this._isActive = true;
        this._center = event.getUILocation();
    }

    private _onTouchMove(event: EventTouch): void {
        if (!this._isActive) return;
        const pos = event.getUILocation();
        const delta = pos.subtract(this._center);
        const len = delta.length();

        if (len > this._maxRadius) {
            delta.multiplyScalar(this._maxRadius / len);
        }

        this._moveDir = delta.normalize();
        // 更新摇杆视觉位置（内圆跟随手指）
    }

    private _onTouchEnd(): void {
        this._isActive = false;
        this._moveDir = Vec2.ZERO.clone();
        // 重置摇杆视觉位置
    }

    get direction(): Vec2 { return this._moveDir; }
    get isActive(): boolean { return this._isActive; }
}
```

## 6. 配置加载器

```typescript
class ConfigManager {
    private _configs: Map<string, any> = new Map();

    async loadAll(): Promise<void> {
        const files = [
            'skills', 'enemies', 'buffs',
            'floorScaling', 'metaUpgrades',
        ];

        await Promise.all(
            files.map(async (name) => {
                const json = await loadJson(`config/${name}`);
                this._configs.set(name, json);
            })
        );
    }

    get<T>(name: string): T {
        return this._configs.get(name) as T;
    }

    getSkills(): SkillDef[] { return this.get('skills'); }
    getEnemies(): EnemyDef[] { return this.get('enemies'); }
    getBuffs(): BuffData[] { return this.get('buffs'); }
}
```

## 7. 存档系统

> 参考：幽灵射手的 `StorageManager` 用 `cc.sys.localStorage`

只存 Meta 状态，局内状态不存（死了就没了）：

```typescript
class SaveManager {
    private static readonly KEY = 'roguelike_meta';

    static save(meta: MetaPlayerState): void {
        const json = JSON.stringify(meta);
        sys.localStorage.setItem(SaveManager.KEY, json);
    }

    static load(): MetaPlayerState {
        const json = sys.localStorage.getItem(SaveManager.KEY);
        if (json) {
            return JSON.parse(json);
        }
        return this._createDefault();
    }

    private static _createDefault(): MetaPlayerState {
        return {
            totalGold: 0,
            highestFloor: 0,
            unlockedSkills: [],
            permanentUpgrades: {},
            totalRuns: 0,
        };
    }
}
```

## 关键设计决策

| 决策 | 理由 |
|------|------|
| JSON 而非 CSV | TypeScript 原生解析，支持嵌套结构，与已有配置风格一致 |
| UIManager 缓存面板 | 避免反复实例化/销毁的性能开销 |
| 只存 Meta 不存局内 | 肉鸽设计哲学：局内状态是临时的 |
| 配置描述中用 `{value}` 占位符 | UI 展示时动态替换，策划改数值不需要改代码 |
